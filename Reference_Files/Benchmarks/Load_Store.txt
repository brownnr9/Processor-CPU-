# STORE AND LOAD BENCH MARK
.data
    my_buffer: .word 0, 0    # Reserve space for two words

.text
    la s0, my_buffer         # 'la' (Load Address) puts the safe address into s0
   
    li t0, 123
    sw t0, 0(s0)             # This will now succeed
   
    addi t0, t0, 2
    sw t0, 4(s0)
   
    lw t1, 0(s0)             # t1 = 123		t2 = 125
    lw t2, 4(s0)

add s1, t1, t2		# s1 = 248
